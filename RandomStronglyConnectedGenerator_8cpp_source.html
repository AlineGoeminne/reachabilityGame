103  k++;
104  }

107  // Comme a et b sont des composantes fortement connexes, on obtient une plus grosse composante fortement connexe

109  vertices[vAID]->addSuccessor(vertices[vBID], generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
110  }
111  if (!vertices[uBID]->hasSuccessor(uAID)) {
112  vertices[uBID]->addSuccessor(vertices[uAID], generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
113  }

116  }
