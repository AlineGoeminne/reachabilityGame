103  Long Long::operator*(const Long& other) const {
104  if (isInfinity() && other.isInfinity()) {
105  if (m_value == other.m_value) {
106  // +inf * +inf = +inf
107  // -inf * -inf = +inf
108  return Long::infinity;
109  }
110  else {
111  // +inf * -inf = -inf
112  // -inf * +inf = -inf
113  return -Long::infinity;
114  }
115  }
116  else if (isInfinity() || other.isInfinity()) {
117  if (std::signbit(m_value) == std::signbit(other.m_value)) {
118  // Même signe => +inf
119  return Long::infinity;
120  }
121  else {
122  return -Long::infinity;
123  }
124  }
125  else {
126  // Pas d'infini => simple multiplication de deux nombres
127  return m_value * other.m_value;
128  }
