49  ReachabilityGame randomGenerator(std::size_t size, std::size_t lowOutgoing, std::size_t upOutgoing, long minWeight, long maxWeight, bool multipleWeights, std::size_t nPlayers, bool sharedTargets, const std::vector<double>& probaPlayers, const std::vector<double>& probaTargets, const std::vector<types::Long>& maximumTargets) {
50  if (probaPlayers.size() != nPlayers || probaTargets.size() != nPlayers || maximumTargets.size() != nPlayers) {
51  throw std::runtime_error("randomGenerator: les tableaux de probabilité doivent arriver une taille identique au nombre de joueurs");
52  }
53  if (lowOutgoing == 0 || upOutgoing == 0) {
54  throw std::runtime_error("randomGenerator: les bornes sur le nombre de noeuds sortants doivent être > 0");
55  }
56  if (lowOutgoing > upOutgoing) {
57  throw std::runtime_error("randomGenerator: la borne inférieure sur le nombre de noeuds sortants ne peut pas être supérieur à la borne supérieure");
58  }
59  if (lowOutgoing > size || upOutgoing > size) {
60  throw std::runtime_error("randomGenerator: les bornes sur le nombre de noeuds sortants ne peuvent pas dépasser le nombre de noeuds");
61  }
62  if (minWeight > maxWeight) {
63  throw std::runtime_error("randomGenerator: le poids minimal doit être inférieur ou égal au poids maximal");
64  }
65  if (std::abs(std::accumulate(probaPlayers.begin(), probaPlayers.end(), 0.) - 1.) > 1E-15) {
66  throw std::runtime_error("randomGenerator: la somme des valeurs de probaPlayers doit être 1.");
67  }
68 
69  // On crée les générateurs aléatoires qui seront utilisés
70  std::default_random_engine generator(std::chrono::system_clock::now().time_since_epoch().count());
71  std::discrete_distribution<std::size_t> playersDistribution(probaPlayers.begin(), probaPlayers.end()); // Sommet appartient au joueur i; i est tiré aléatoirement selon probaPlayers
72  std::vector<std::bernoulli_distribution> targetsDistributions; // Pour chaque joueur, probabilité qu'un sommet soit une cible
73  std::uniform_int_distribution<std::size_t> numberOfNeighborsDistribution(lowOutgoing, upOutgoing); // Distribution pour le nombre d'arcs sortants
74  std::uniform_int_distribution<std::size_t> neighborDistribution(0, size-1); // Distribution pour les sommets des arcs
75  auto weightDistribution = constructWeightDistribution(minWeight, maxWeight);
76 
77  // On crée les joueurs
78  std::vector<Player> players;
79  for (std::size_t i = 0 ; i < nPlayers ; i++) {
80  players.emplace_back(i); // On construit le joueur en lui donnant uniquement son ID
81  targetsDistributions.emplace_back(probaTargets[i]); // On construit le générateur aléatoire (booléen) pour décider si un sommet est une cible pour le joueur i
82  }
83 
84  // On crée tous les sommets
85  std::vector<Vertex::Ptr> vertices(size, nullptr);
86  for (std::size_t i = 0 ; i < size ; i++) {
87  // On tire aléatoirement le joueur qui possède ce noeud
88  std::size_t player = playersDistribution(generator);
89 
90  vertices[i] = std::make_shared<Vertex>(i, player, nPlayers);
91 
92  players[player].addVertex(vertices[i]);
93 
94  // On décide aléatoirement pour chaque joueur si le sommet est une cible
95  for (std::size_t j = 0 ; j < nPlayers ; j++) {
96  // On vérifie aussi que le joueur peut encore avoir une nouvelle cible
97  if (maximumTargets[j] > players[j].getGoals().size() && targetsDistributions[j](generator)) {
98  players[j].addGoal(vertices[i]);
99  vertices[i]->addTargetFor(j);
100 
101  if (!sharedTargets) {
102  // On ne peut partager les cibles. Donc, on arrête
103  break;
104  }
105  }
106  }
107  }
108 
109  // On va vérifier que chaque joueur a bien au moins une cible. Si non, on va lui en donner une de force en négligeant sharedTargets
110  for (Player &p : players) {
111  if (p.getGoals().size() == 0) {
112  p.addGoal(vertices[neighborDistribution(generator)]);
113  }
114  }
115 
116  // On crée les arcs
117  for (std::size_t i = 0 ; i < size ; i++) {
118  Vertex::Ptr v = vertices[i];
119 
120  // On tire le nombre d'arcs sortants
121  std::size_t e = numberOfNeighborsDistribution(generator);
122 
123  while (e-- > 0) {
124  // On détermine le voisin
125  Vertex::Ptr u = vertices[neighborDistribution(generator)];
126  // On ne veut pas avoir deux arcs v->u
127  while (v->hasSuccessor(u->getID())) {
128  u = vertices[neighborDistribution(generator)];
129  }
130 
131  v->addSuccessor(u, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
132  }
133  }
134 
135  Graph graph(vertices, nPlayers);
136 
137  ReachabilityGame game(graph, vertices[0], players);
138 
139  return game;
140  }
