52 void Path::addStep(std::shared_ptr<const Vertex> step) {
53  // On vérifie d'abord que l'arc entre last et step existe
54  std::shared_ptr<const Vertex> last = m_path.back();
55  Vertex::Edge e = last->getSuccessor(step->getID());
56  // weak_ptr !
57  if (!e.first.lock()) {
58  throw InvalidPath("Il n'existe pas d'arc entre le dernier sommet (" + std::to_string(last->getID()) + ") du chemin et l'étape donnée (" + std::to_string(step->getID()) + ")");
59  }
60  // S'il existe, on ajoute les coûts sur l'arc aux coûts déjà calculés
61  const std::vector<Long>& costs = e.second;
62  for (std::size_t i = 0 ; i < m_costs.size() ; i++) {
63  // On vérifie que le joueur i n'a pas déjà atteint une cible
64  if (!m_costs[i].first) {
65  m_costs[i].second += costs[i];
66  }
67  }
68  // On met à jour les joueurs qui ont atteint une cible
69  for (const unsigned int &player : step->getTargetPlayers()) {
70  m_costs[player].first = true;
71  }
72  // Finalement, on enregistre le pas
73  m_path.push_back(step);
74 }

80 bool Path::isANashEquilibrium(std::unordered_set<unsigned int> playersAlreadyTested) const {
81  std::size_t nPlayers = m_costs.size();
82  std::vector<Long> epsilon(nPlayers, 0); // Poids du chemin jusqu'au noeud courant
83  bool nash = true;
84 
85  // unordered_...::find ont une complexité de O(1) en moyenne
86  std::unordered_set<unsigned int> visitedPlayers; // Ensemble des joueurs ayant déjà atteint une de leurs cibles
87  std::unordered_map<unsigned int, std::vector<Long>> coalitions; // A un joueur, on associe les valeurs de la coalition contre lui
88 
89  for (auto itr = m_path.begin() ; nash && itr != m_path.end() ; ++itr) {
90  const std::shared_ptr<const Vertex> current = *itr;
91 
92  auto& players = current->getTargetPlayers();
93  visitedPlayers.insert(players.begin(), players.end()); // Equivalent de l'union
94 
95  if (itr != m_path.begin()) {
96  // Si on n'est pas au premier sommet, on incrémente epsilon
97  const std::shared_ptr<const Vertex> prev = *std::prev(itr); // On récupère le vertex précédent
98  std::vector<Long> weights = prev->getWeights(current->getID());
99 
100  for (std::size_t i = 0 ; i < epsilon.size() ; i++) {
101  epsilon[i] += weights[i];
102  }
103  }
104 
105  unsigned int player = current->getPlayer();
106 
107  if (playersAlreadyTested.find(player) == playersAlreadyTested.end() && visitedPlayers.find(player) == visitedPlayers.end()) {
108  // Si le joueur actuel n'a pas déjà atteint une cible (et qu'il n'a pas déjà été testé), on vérifie si c'est bien un EN
109 
110  Long val(0);
111 
112  if (coalitions.find(player) == coalitions.end()) {
113  // On ne connait pas encore les valeurs
114  MinMaxGame minmax = MinMaxGame::convert(m_game, player);
115 
116  std::vector<Long> values = minmax.getValues(m_game.getPlayers()[player].getGoals());
117  coalitions[player] = values;
118 
119  val = values[current->getID()];
120  }
121  else {
122  // On connait déjà les valeurs
123  val = coalitions[player][current->getID()];
124  }
125 
126  if (!respectProperty(val, epsilon, player)) {
127  nash = false;
128  }
129  }
130  }
131 
132  return nash;
133 }

147 bool operator==(const Path& a, const Path &b) {
148  if (a.size() != b.size()) {
149  // Pas le même nombre de pas
150  return false;
151  }
152 
153  auto itrA = a.m_path.begin();
154  auto itrB = b.m_path.begin();
155 
156  for (; itrA != a.m_path.end() && itrB != b.m_path.end() ; ++itrA, ++itrB) {
157  const std::shared_ptr<const Vertex> va = *itrA;
158  const std::shared_ptr<const Vertex> vb = *itrB;
159  if (*va != *vb) {
160  // Les chemins ont des pas différents
161  return false;
162  }
163  }
164  // Même longueur et mêmes pas
165  return true;
166 }
