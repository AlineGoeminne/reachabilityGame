47 MinMaxGame MinMaxGame::convert(const ReachabilityGame &game, unsigned int minPlayer) {
48  // TODO : si trop lent, trouver autre moyen
49  std::vector<Vertex::Ptr> newVertices;
50  std::unordered_set<Vertex::Ptr> minVertices, maxVertices;
51  std::unordered_set<Vertex::Ptr> minGoals;
52  const Graph &graph = game.getGraph();
53  // On commence par créer les nouveaux sommets
54  for (auto itr = graph.cbegin() ; itr != graph.cend() ; ++itr) {
55  const Vertex::Ptr vertex = *itr;
56  // Même ID
57  // On répartit entre Min et Max
58  DijVertex::Ptr newV;
59  if (vertex->getPlayer() == minPlayer) {
60  newV = std::make_shared<DijVertex>(vertex->getID(), MIN, 2);
61  minVertices.insert(newV);
62  }
63  else {
64  newV = std::make_shared<DijVertex>(vertex->getID(), MAX, 2);
65  maxVertices.insert(newV);
66  }
67 
68  newVertices.push_back(std::move(newV));
69  }
70 
71  // On copie les successeurs et prédecesseurs
72  for (std::size_t i = 0 ; i < graph.size() ; i++) {
73  for (auto &e : *graph.getVertices()[i]) {
74  // e.first = ID du vertex
75  // e.second.second = poids sur l'arc; on ne garde que le poids pour MIN
76  newVertices[i]->addSuccessor(newVertices[e.first], e.second.second[minPlayer]);
77  }
78  }
79 
80  // Pour finir, on crée les deux nouveaux joueurs
81  Player min(MIN, minVertices, minGoals), max(MAX, maxVertices, {});
82 
83  // Les poids sur les arcs sont les mêmes
84  Graph g(newVertices, graph.getMaxWeights());
85 
86  return MinMaxGame(g, game.getInit(), min, max);
87 }

126 void MinMaxGame::dijkstraMinMax(const std::unordered_set<Vertex::Ptr> &goals) {
127  initQ(goals);
128  initS(goals);
129 
130  while (!m_Q.empty()) {
131  auto s = m_Q.top();
132  Successor successor = s->S.top();
133 
134  if (successor.cost == Long::infinity) {
135  m_Q.pop();
136  }
137  else if (goals.find(s) != goals.end() || s->getPlayer() == MIN || s->nSuccessors == 1) {
138  // Si c'est une cible ou si le sommet appartient à Min ou si le nombre de successeurs est 1, on relaxe le sommet
139  m_Q.pop();
140  for (auto itr = s->beginPredecessors() ; itr != s->endPredecessors() ; itr++) {
141  relax(s, itr->second);
142  }
143  }
144  else {
145  // Le sommet appartient à Max et nSuccessors > 1
146  // On bloque la plus petite valeur
147  s->S.pop();
148  Successor newSucc = s->S.top();
149  m_Q.updateKeyPointer<Long>(s, newSucc.cost);
150  s->nSuccessors--;
151  }
152  }
153 }

155 void MinMaxGame::initQ(const std::unordered_set<Vertex::Ptr>& goals) {
156  m_Q = DynamicPriorityQueue<DijVertex::Ptr, CompareDijVertex>();
157  auto &vertices = getGraph().getVertices();
158  for (std::size_t i = 0 ; i < vertices.size() ; i++) {
159  DijVertex::Ptr v = std::static_pointer_cast<DijVertex>(vertices[i]);
160  if (goals.find(v) != goals.end()) {
161  // v est un goal
162  v->d = 0;
163  }
164  else {
165  v->d = Long::infinity;
166  }
167  m_Q.push(v);
168  }
169 }

188 void MinMaxGame::relax(DijVertex::Ptr s, const Vertex::Edge& edge) {
189  DijVertex::Ptr p = std::dynamic_pointer_cast<DijVertex>(edge.first.lock());
190  const Successor &succ = s->S.top();
191  Long pVal = edge.second[MIN] + succ.cost;
192  const Successor &old = p->S.top();
193 
194  if (pVal < old.cost) {
195  m_Q.updateKeyPointer(p, pVal);
196  if (p->getPlayer() == MIN) {
197  Successor newSucc;
198  newSucc.cost = pVal;
199  p->S = std::priority_queue<Successor, std::vector<Successor>, Successor>();
200  p->S.push(newSucc);
201  }
202  }
203 
204  if (p->getPlayer() == MAX) {
205  Successor succ;
206  succ.cost = pVal;
207  p->S.push(succ);
208  }
209 }
