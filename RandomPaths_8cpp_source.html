30 Path generatePath(const ReachabilityGame &game, std::size_t length) {
31  std::shared_ptr<const Vertex> current = game.getInit();
32  std::default_random_engine generator(std::chrono::system_clock::now().time_since_epoch().count());
33 
34  Path path(game, current);
35 
36  while(length--) {
37  // On va tirer le successeur au hasard
38  std::uniform_int_distribution<std::size_t> successorChoice(0, current->getNumberSuccessors() - 1);
39 
40  std::size_t succIndex = successorChoice(generator);
41 
42  std::shared_ptr<const Vertex> succ = nullptr;
43 
44  for (auto s = current->cbegin() ; s != current->cend() ; s++) {
45  if (succIndex-- == 0) {
46  succ = s->second.first.lock();
47  break;
48  }
49  }
50 
51  path.addStep(succ);
52  current = succ;
53  }
54 
55  return path;
56 }

63 const Path& filterBest(const std::vector<Path> &paths, std::size_t nPlayers) {
64  std::size_t bestCostIndex = 0;
65  std::size_t bestReachedIndex = 0;
66  types::Long valueBestCost = types::Long::infinity;
67  types::Long valueBestReached = 0;
68  types::Long partialValueBestCost = types::Long::infinity;
69  bool allPlayersEN = false;
70 
71  for (std::size_t i = 0 ; i < paths.size() ; i++) {
72  const Path &p = paths[i];
73  const auto &costs = p.getCosts();
74 
75  types::Long sumCost = 0; // Les coûts totaux
76  types::Long partialCost = 0; // La somme de coûts pour les joueurs qui ont vu leur cible
77  std::size_t nReached = 0; // Le nombre de joueurs qui ont vu une cible
78  for (const auto &cost : costs) {
79  if (cost.first) {
80  nReached++;
81  partialCost += cost.second;
82  }
83  sumCost += cost.second;
84  }
85 
86  if (nReached != nPlayers && valueBestCost > sumCost) {
87  // Tout le monde n'a pas vu une cible mais on a un coût inférieur à celui trouvé précédemment
88  bestCostIndex = i;
89  valueBestCost = sumCost;
90  }
91  else if (valueBestReached < nReached) {
92  // On voit plus de monde qu'avant
93  bestReachedIndex = i;
94  valueBestReached = nReached;
95  if (nPlayers == nReached) {
96  // On a vu tout le monde
97  allPlayersEN = true;
98  }
99  }
100  else if (valueBestReached == nReached && partialCost < partialValueBestCost) {
101  // On voit autant de monde que l'EN déjà retenu mais les coûts sont moindres
102  bestReachedIndex = i;
103  partialValueBestCost = partialCost;
104  }
105  }
106 
107  if (allPlayersEN) {
108  return paths[bestReachedIndex];
109  }
110  else {
111  return paths[bestCostIndex];
112  }
113 }

116  Path randomPath(const ReachabilityGame &game, std::size_t nPaths) {
117  std::vector<Path> paths;
118  paths.reserve(nPaths);
119 
120  for (std::size_t i = 0 ; i < nPaths ; i++) {
121  Path p = generatePath(game, game.getMaxLength());
122  if (p.isANashEquilibrium()) {
123  paths.push_back(p);
124  }
125  }
126 
127  if (paths.size() == 0) {
128  throw NoENGenerated("Pas d'équilibre de Nash généré");
129  }
130 
131  return filterBest(paths, game.getMaxLength());
132  }
