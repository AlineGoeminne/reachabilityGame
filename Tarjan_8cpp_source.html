38  indices[v] = index;
39  lowLink[v] = index;
40  index++;
41 
42  stack.push(v);
43  onStack[v] = true;
44

47  for (const auto &e : *vertices[v]) {
48  unsigned int w = e.first;
49 
50  if (indices[w] == 0) {

52  components.insert(compo.begin(), compo.end());
53  lowLink[v] = std::min(lowLink[v], lowLink[w]);
54  }

56  lowLink[v] = std::min(lowLink[v], indices[w]);
57  }
58  }

62  unsigned int w = stack.top();
63  stack.pop();
64  onStack[w] = false;
65  component.insert(w);
66

69  stack.pop();
70  onStack[w] = false;
71  component.insert(w);
72  }
73

75  }
