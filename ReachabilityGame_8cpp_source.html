27 ReachabilityGame::ReachabilityGame(Graph graph, Vertex::Ptr init, const std::vector<Player>& players) :
28  Game(graph, init),
29  m_players(players),
30  m_maxWeightsPath(players.size())
31  {
32  for (std::size_t i = 0 ; i < players.size() ; i++) {
33  // Le poids maximal pour un chemin est (|Pi| + 1) * |V| * max(|w_i|)
34  m_maxWeightsPath[i] = Long(getMaxLength()) * getGraph().getMaxWeights()[i];
35  }
36 }

90 Long ReachabilityGame::AStartPositive(const Node::Ptr& node, const CostsMap &costsMap) {
91  const std::vector<Long> &epsilon = node->state.epsilon;
92  // g(n) = Coût vers les cibles déjà atteintes + coûts partiels (joueurs qui n'ont pas encore atteints une cible)
93  Long g_n = node->state.RP;
94  for (unsigned int notReached : node->state.notVisitedPlayers) {
95  g_n += epsilon[notReached];
96  }
97 
98  Long h_n = 0;
99  for (unsigned int notReached : node->state.notVisitedPlayers) {
100  Long shortest = Long::infinity;
101  for (const Vertex::Ptr goal : m_players[notReached].getGoals()) {
102  auto itr = costsMap.find(goal);
103  const CostsForATarget& target = itr->second;
104  shortest = std::min(shortest, target[node->path.getLast()->getID()]);
105  }
106  h_n += std::min(shortest, m_maxWeightsPath[notReached] - epsilon[notReached]);
107  }
108 
109  return g_n + h_n;
110 }

112 std::size_t ReachabilityGame::percentageOfReachableVertices() const {
113  std::size_t nReachable = 0;
114  std::queue<Vertex::Ptr> queue;
115  std::unordered_set<Vertex::Ptr> explored;
116 
117  queue.push(m_init);
118  explored.insert(m_init);
119 
120  while (!queue.empty()) {
121  const Vertex::Ptr v = queue.front();
122  queue.pop();
123  nReachable++;
124 
125  for (auto itr : *v) {
126  const Vertex::Ptr u = itr.second.first.lock();
127 
128  // Si on n'a pas encore exploré ce noeud
129  if (explored.find(u) == explored.end()) {
130  queue.push(u);
131  explored.insert(u);
132  }
133  }
134  }
135 
136  return nReachable * 100./getGraph().size();
137 }
