72  ReachabilityGame randomTreeLikeGenerator(std::size_t size, std::size_t lowBranchingFactor, std::size_t upBranchingFactor, double probaSelf, double probaSameDepth, double probaSkipping, double probaClimbing, long minWeight, long maxWeight, bool multipleWeights, std::size_t nPlayers, bool sharedTargets, const std::vector<double>& probaPlayers, const std::vector<double> &probaTargets, const std::vector<types::Long>& maximumTargets) {
73  if (probaPlayers.size() != nPlayers || probaTargets.size() != nPlayers || maximumTargets.size() != nPlayers) {
74  throw std::runtime_error("randomTreeLikeGenerator: les tableaux de probabilité doivent arriver une taille identique au nombre de joueurs");
75  }
76  if (lowBranchingFactor == 0 || lowBranchingFactor == 0) {
77  throw std::runtime_error("randomTreeLikeGenerator: les bornes sur le facteur de branchement doivent être > 0");
78  }
79  if (lowBranchingFactor > upBranchingFactor) {
80  throw std::runtime_error("randomTreeLikeGenerator: la borne inférieure sur le facteur de branchement ne peut pas être supérieur à la borne supérieure");
81  }
82  if (lowBranchingFactor > size || upBranchingFactor > size) {
83  throw std::runtime_error("randomTreeLikeGenerator: les bornes sur le facteur de branchement ne peuvent pas dépasser le nombre de noeuds");
84  }
85  if (minWeight > maxWeight) {
86  throw std::runtime_error("randomTreeLikeGenerator: le poids minimal doit être inférieur ou égal au poids maximal");
87  }
88  if (!equal(std::accumulate(probaPlayers.begin(), probaPlayers.end(), 0.), 1.)) {
89  throw std::runtime_error("randomTreeLikeGenerator: la somme des valeurs de probaPlayers doit être 1");
90  }
91  if (!(lowerOrEqual(0, probaSelf) && lowerOrEqual(probaSelf, 1))) {
92  throw std::runtime_error("randomTreeLikeGenerator: probaSelf doit être dans [0, 1]");
93  }
94  if (!(lowerOrEqual(0, probaSkipping) && lowerOrEqual(probaSkipping, 1))) {
95  throw std::runtime_error("randomTreeLikeGenerator: probaSkipping doit être dans [0, 1]");
96  }
97  if (!(lowerOrEqual(0, probaClimbing) && lowerOrEqual(probaClimbing, 1))) {
98  throw std::runtime_error("randomTreeLikeGenerator: probaClimbing doit être dans [0, 1]");
99  }
100  if (!(lowerOrEqual(0, probaSameDepth) && lowerOrEqual(probaSameDepth, 1))) {
101  throw std::runtime_error("randomTreeLikeGenerator: probaSameDepth doit être dans [0, 1]");
102  }
103  for (double d : probaTargets) {
104  if (!(lowerOrEqual(0, d) && lowerOrEqual(d, 1))) {
105  throw std::runtime_error("randomTreeLikeGenerator: les probabilités dans probaTargets doivent être dans [0, 1]");
106  }
107  }
108 
109  // On crée les générateurs aléatoires qui seront utilisés
110  std::default_random_engine generator(std::chrono::system_clock::now().time_since_epoch().count());
111  std::discrete_distribution<std::size_t> playersDistribution(probaPlayers.begin(), probaPlayers.end()); // Sommet appartient au joueur i; i est tiré aléatoirement selon probaPlayers
112  std::vector<std::bernoulli_distribution> targetsDistributions; // Pour chaque joueur, probabilité qu'un sommet soit une cible
113  std::uniform_int_distribution<std::size_t> branchingFactorDistribution(lowBranchingFactor, upBranchingFactor); // Distribution pour le facteur de branchement
114  auto weightDistribution = constructWeightDistribution(minWeight, maxWeight);
115  std::bernoulli_distribution selfDistribution(probaSelf); // Distribution pour savoir si on peut générer un arc de v à v
116  std::bernoulli_distribution sameDepthDistribution(probaSameDepth); // Distribution pour savoir si on peut générer un arc entre deux sommets à la même profondeur
117  std::bernoulli_distribution skippingDistribution(probaSkipping); // Distribution pour savoir si on peut générer un arc entre un sommet et un autre plus profond qui n'est pas son enfant
118  std::bernoulli_distribution climbingDistribution(probaClimbing); // Distribution pour savoir si on peut générer un arc qui monte dans l'arbre
119  std::uniform_int_distribution<std::size_t> forceTargetDistribution(0, size-1); // Distribution pour décider quel sommet devient une cible quand il faut en forcer une pour un joueur
120 
121  // On crée les joueurs
122  std::vector<Player> players;
123  for (std::size_t i = 0 ; i < nPlayers ; i++) {
124  players.emplace_back(i); // On construit le joueur en lui donnant uniquement son ID
125  targetsDistributions.emplace_back(probaTargets[i]); // On construit le générateur aléatoire (booléen) pour décider si un sommet est une cible pour le joueur i
126  }
127 
128  // On va créer les sommets
129  std::vector<TreeVertex::Ptr> vertices(size, nullptr);
130  for (std::size_t i = 0 ; i < size ; i++) {
131  // On choisit le joueur
132  std::size_t player = playersDistribution(generator);
133  vertices[i] = std::make_shared<TreeVertex>(i, player, nPlayers);
134 
135  players[player].addVertex(vertices[i]);
136 
137  // On décide si le sommet est une cible
138  for (std::size_t j = 0 ; j < nPlayers ; j++) {
139  if (players[j].getGoals().size() < maximumTargets[j] && targetsDistributions[j](generator)) {
140  players[j].addGoal(vertices[i]);
141  vertices[i]->addTargetFor(j);
142 
143  if (!sharedTargets) {
144  // On s'arrête parce qu'on ne peut pas partager une cible
145  break;
146  }
147  }
148  }
149  }
150 
151  // On vérifie que tous les joueurs ont une cible
152  for (Player &p : players) {
153  if (p.getGoals().size() == 0) {
154  p.addGoal(vertices[forceTargetDistribution(generator)]);
155  }
156  }
157 
158  vertices[0]->m_depth = 0;
159  std::queue<TreeVertex::Ptr> queue;
160  queue.push(vertices[0]);
161  std::size_t nextFreeID = 1;
162 
163  while (!queue.empty() && nextFreeID < size) {
164  TreeVertex::Ptr root = queue.front();
165  queue.pop();
166 
167  std::size_t e = branchingFactorDistribution(generator);
168 
169  for (std::size_t i = 0 ; i < e && nextFreeID < size ; i++) {
170  TreeVertex::Ptr child = vertices[nextFreeID++];
171  child->m_depth = root->m_depth + 1;
172  root->addSuccessor(child, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
173 
174  queue.push(child);
175  }
176  }
177 
178  for (TreeVertex::Ptr v : vertices) {
179  for (TreeVertex::Ptr u : vertices) {
180  if (v->hasSuccessor(u->getID())) {
181  continue;
182  }
183  if (v->getID() == u->getID() && selfDistribution(generator)) {
184  // Même sommet
185  v->addSuccessor(u, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
186  }
187  else if (u->m_depth > v->m_depth && skippingDistribution(generator)) {
188  // On veut descendre dans l'arbre
189  v->addSuccessor(u, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
190  }
191  else if (u->m_depth == v->m_depth && sameDepthDistribution(generator)) {
192  // Frères/cousins
193  v->addSuccessor(u, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
194  }
195  else if (u->m_depth < v->m_depth && climbingDistribution(generator)) {
196  // On veut remonter dans l'arbre
197  v->addSuccessor(u, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
198  }
199  }
200 
201  // On vérifie qu'on n'est pas dans un cul-de-sac
202  if (v->getNumberSuccessors() == 0) {
203  // Sinon, on ajoute un arc de v à v
204  v->addSuccessor(v, generateWeights(nPlayers, generator, weightDistribution, multipleWeights));
205  }
206  }
207 
208  std::vector<Vertex::Ptr> baseVertices(vertices.begin(), vertices.end());
209  Graph graph(baseVertices, nPlayers);
210 
211  ReachabilityGame game(graph, baseVertices[0], players);
212 
213  return game;
214  }
