63  Path bestFirstSearch(const ReachabilityGame& game, const heuristicSignature& heuristic, Long allowedTime) {
64  std::size_t nPlayers = game.getGraph().getNumberPlayers();
65 
66  CostsMap costsMap = computeAllDijkstra(game);
67 
68  // On initialise le premier noeud de l'exploration
69  Path path(game, game.getInit());
70  Node::Ptr init = std::make_shared<Node>(nPlayers, path);
71  for (unsigned int player : game.getInit()->getTargetPlayers()) {
72  init->state.notVisitedPlayers.erase(player);
73  }
74  init->pathCost = heuristic(init, costsMap);
75 
76  // On construit la frontière avec le noeud créé
77  std::priority_queue<Node::Ptr, std::vector<Node::Ptr>, CompareNodes> frontier;
78  frontier.push(init);
79 
80  std::unordered_set<unsigned int> allPlayers;
81  for (unsigned int i = 0 ; i < nPlayers ; i++) {
82  allPlayers.insert(i);
83  }
84 
85  auto start = std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now().time_since_epoch()).count();
86 
87  while(std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now().time_since_epoch()).count() - start < allowedTime) {
88  if (frontier.empty()) {
89  throw EmptyFrontier("La frontière est vide");
90  }
91 
92  Node::Ptr currentNode = frontier.top();
93  frontier.pop();
94 
95  if (currentNode->state.notVisitedPlayers.size() == 0) {
96  // Tout le monde a vu une cible. On a donc un équilibre de Nash (si exploration optimale)
97  return currentNode->path;
98  }
99  else if (currentNode->path.size() == game.getMaxLength()) {
100  // On a atteint la longueur maximale
101  // Il se peut que ce soit un équilibre de Nash
102  // On va vérifier pour les joueurs qui n'ont pas encore atteint un objectif
103 
104  if (currentNode->path.isANashEquilibrium(currentNode->state.notVisitedPlayers)) {
105  // On a un EN
106  return currentNode->path;
107  }
108  }
109  else {
110  const std::shared_ptr<const Vertex> last = currentNode->path.getLast();
111 
112  // On va itérer sur chaque successeur du dernier sommet du chemin
113  for (auto succEdge = last->cbegin() ; succEdge != last->cend() ; succEdge++) {
114  const Vertex::Ptr succ = succEdge->second.first.lock();
115  const std::vector<Long>& w = succEdge->second.second;
116 
117  // On copie le noeud et on ajoute un pas
118  Node::Ptr newNode = std::make_shared<Node>(currentNode);
119  newNode->path.addStep(succ);
120 
121  // On met à jour les coûts en ajoutant le coût de l'arc emprunté
122  for (std::size_t i = 0 ; i < nPlayers ; i++) {
123  newNode->state.epsilon[i] = w[i] + currentNode->state.epsilon[i];
124  }
125 
126  if (succ->isTarget()) {
127  std::unordered_set<unsigned int> newReached;
128  for (unsigned int p : newNode->state.notVisitedPlayers) {
129  if (succ->isTargetFor(p)) {
130  newReached.insert(p);
131  }
132  }
133 
134  if (newReached.size() != 0) {
135  // On a des joueurs qui atteignent une cible pour la première fois
136  bool nash = true;
137  // On vérifie si on a un équilibre de Nash pour chaque joueur
138  for (unsigned int p : newReached) {
139  allPlayers.erase(p);
140  if (!newNode->path.isANashEquilibrium(allPlayers)) {
141  nash = false;
142  }
143  allPlayers.insert(p);
144  }
145  // Si oui, on va ajouter un nouveau noeud à la frontière
146  if (nash) {
147  for (unsigned int p : newReached) {
148  newNode->state.notVisitedPlayers.erase(p);
149  newNode->state.RP += newNode->state.epsilon[p];
150  }
151  newNode->pathCost = heuristic(newNode, costsMap);
152  frontier.push(newNode);
153  }
154  }
155  else {
156  newNode->pathCost = heuristic(newNode, costsMap);
157  frontier.push(newNode);
158  }
159  }
160  else {
161  newNode->pathCost = heuristic(newNode, costsMap);
162  frontier.push(newNode);
163  }
164  }
165  }
166  }
167  throw OutOfTime("L'exploration s'est achevée par manque de temps");
168  }
